document.getElementById('scrollButton').addEventListener('click', function() {
    const destino = document.getElementById('destino');
    
    // Calcula a posição Y que queremos alcançar
    const destinoY = destino.getBoundingClientRect().top + window.pageYOffset;
    
    // Define a duração da animação em milissegundos (ex: 1500ms = 1.5 segundos)
    const duracao = 1500; 

    // Função de animação personalizada
    function animateScroll(duration) {
        const start = window.pageYOffset;
        const startTime = performance.now();

        function scrollStep(timestamp) {
            const currentTime = timestamp || performance.now();
            const progress = Math.min(1, (currentTime - startTime) / duration);
            
            // Função de easing para um movimento suave (ease out quad)
            const easeOutProgress = progress * (2 - progress); 

            // Move a tela
            window.scrollTo(0, start + (destinoY - start) * easeOutProgress);

            // Continua a animação se não tiver terminado
            if (progress < 1) {
                window.requestAnimationFrame(scrollStep);
            }
        }
        
        window.requestAnimationFrame(scrollStep);
    }

    animateScroll(duracao);
});

// 1. Seleciona todos os links que têm a classe 'js-scroll-link'
const scrollLinks = document.querySelectorAll('.js-scroll-link');

// 2. Itera sobre cada link e adiciona um "ouvinte" de clique
scrollLinks.forEach(link => {
    link.addEventListener('click', function(event) {
        // Previne o comportamento padrão do link (que é pular instantaneamente)
        event.preventDefault();

        // Obtém o ID do destino a partir do atributo href (ex: "#mainPJT")
        const targetId = this.getAttribute('href');
        const destino = document.querySelector(targetId);

        if (destino) {
            const destinoY = destino.getBoundingClientRect().top + window.pageYOffset;
            const duracao = 1000; // 1 segundo de animação

            // Executa a função de animação personalizada
            animateScroll(destinoY, duracao);
        }
    });
});

// A função de animação é a mesma de antes:
function animateScroll(destinoY, duration) {
    const start = window.pageYOffset;
    const startTime = performance.now();

    function scrollStep(timestamp) {
        const currentTime = timestamp || performance.now();
        const progress = Math.min(1, (currentTime - startTime) / duration);
        const easeOutProgress = progress * (2 - progress); 
        
        window.scrollTo(0, start + (destinoY - start) * easeOutProgress);

        if (progress < 1) {
            window.requestAnimationFrame(scrollStep);
        }
    }
    
    window.requestAnimationFrame(scrollStep);
}
